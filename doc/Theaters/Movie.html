def self.create(fields)
  case fields[:year].to_i
  when 1900..1945
    Movie::Ancient
  when 1946..1968
    Movie::Classic
  when 1969..2000
    Movie::Modern
  when 2001..Time.new.year
    Movie::New
  end.new(fields)
end

def matches?(filter, value)
  if send(filter).is_a? Array
    send(filter).any? { |v| value === v }
  else
    value === send(filter)
  end
end
