def show(params = {})
  movie =
    if block_given?
      @films.select { |film| yield film }.sample
    else
      random_by_stars(apply_filters(params))
    end
  raise NoMoney.new(movie.title, movie.price, money) if money < movie.price

  @money -= movie.price
  self.film = movie
  self.start_time = Time.now

  "Now showing: #{film}"
end
